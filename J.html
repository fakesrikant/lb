    # importing the modules
    import PyPDF2
    import pyttsx3
    
    # path of the PDF file
    Audio_Book = open('E://Science//Motion.pdf','rb')
    
    # creating a PdfFileReader object
    pdfReader = PyPDF2.PdfFileReader(Audio_Book)
    
    # the page with which you want to start
    # this will read the page of 25th page.
    Book_Page = pdfReader.numPages
    Book_Speaker = pyttsx3.init()
    Choose_Page = pdfReader.getPage(0)
    
    # extracting the text from the PDF
    Book_text = Choose_Page.extractText()
    
    # reading the text
    Book_Speaker.say(Book_text)
    Book_Speaker.runAndWait()
